green := [32m
reset := [0m
cyan := [36m

setup: $(projectmpath)\tmp $(projectmpath)\scripts $(projectmpath)/scripts/sys.cmd $(projectmpath)/scripts/tasks.cmd $(projectmpath)/scripts/proj.cmd 
	@echo $(green)Info: proj.cmd is up to date$(reset)

$(projectmpath)\tmp: 
	mkdir $(projectpath)\tmp

$(projectmpath)\scripts :
	mkdir $(projectpath)\scripts

$(projectmpath)/scripts/proj.cmd: $(projectmpath)/scripts/sys.cmd $(projectpath)/scripts/tasks.cmd scripts/func.cmd
	@echo $(cyan)Updating proj.cmd$(reset)
	@copy $(projectpath)\scripts\tasks.cmd+$(projectpath)\scripts\sys.cmd+scripts\func.cmd $(projectpath)\scripts\proj.cmd > nul


$(projectmpath)/scripts/sys.cmd: setup/xrun.ini
	@echo $(cyan)Updating sys.cmd$(reset)
	@call "$(ccw32)" -u -b -q -n -t "scripts\setup2.cct" -o "$(projectpath)\scripts\sys.cmd" "setup\xrun.ini"

$(projectpath)/scripts/tasks.cmd: $(projectmpath)\project.txt
	@echo $(cyan)Updating task.cmd$(reset)
	@call "$(ccw32)" -u -b -q -n -t "scripts\setup2.cct" -o "$(projectpath)\scripts\tasks.cmd" "$(projectpath)\project.txt"
