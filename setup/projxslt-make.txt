green := [32m
reset := [0m

	@echo $(green)Info: project.xslt is up to date$(reset)

	@echo $(cyan)Rebuilding: project.xslt$(reset)
	@rem echo java -jar "$(xrunnerpath)\tools\saxon\saxon9he.jar" -o:"project.xslt" "tmp\proj-var.xml" "$(xrunnerpath)\scripts\projectvariables-v2.xslt" projectpath=${CURDIR} USERPROFILE=$(USERPROFILE)
	@java -jar "$(saxon)" -o:"scripts\project.xslt" "tmp\proj-var.xml" "$(xrunnerpath)\scripts\projectvariables-v2.xslt" projectpath=${CURDIR} USERPROFILE=$(USERPROFILE)

	@"D:\programs\xrunner2\tools\cct\Ccw64.exe" -u -b -q -n -t "$(xrunnerpath)\scripts\proj-var.cct" -o "tmp\proj-var.xml" "project.txt"
